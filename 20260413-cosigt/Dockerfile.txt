# Dockerfile for Cosigt execution environment
# Replicates: micromamba create -n smk7324app132 ...
# Reference: https://davidebolo1993.github.io/cosigtdoc/setup/setup.html
#
# Singularity compatible: all paths are world-readable and
# not bound to a specific user, since Singularity ignores
# USER directives and runs as the invoking host user.

FROM mambaorg/micromamba:1.5.10 AS build

USER root

# System dependencies required by Apptainer and Snakemake
RUN apt-get update && apt-get install -y --no-install-recommends \
    git \
    curl \
    wget \
    ca-certificates \
    squashfs-tools \
    fuse2fs \
    fuse-overlayfs \
    uidmap \
    && rm -rf /var/lib/apt/lists/*

# Create the conda environment with pinned versions
# Use root so the environment is installed under /opt/conda (world-accessible)
RUN micromamba create -y \
    -n smk7324app132 \
    bioconda::snakemake=7.32.4 \
    conda-forge::apptainer=1.3.2 \
    conda-forge::cookiecutter=2.6.0 \
    conda-forge::gdown \
    && micromamba clean -afy

# Clone cosigt pipeline to a shared location
RUN git clone https://github.com/davidebolo1993/cosigt /opt/cosigt

# Ensure all installed files are readable/executable by any user
RUN chmod -R a+rX /opt/conda /opt/cosigt

# Put the conda environment on PATH directly so it works
# regardless of which user runs the container (no `activate` needed)
ENV PATH=/opt/conda/envs/smk7324app132/bin:$PATH
ENV MAMBA_DEFAULT_ENV=smk7324app132

WORKDIR /opt/cosigt/cosigt_smk

CMD ["/bin/bash"]
