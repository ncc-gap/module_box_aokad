    git \
    wget \
    bzip2 \
    make \
    gcc \
    pigz \
    zlib1g-dev \
    libbz2-dev \
    liblzma-dev \
    libcurl4-openssl-dev \
    libssl-dev \
    python3 \
    python3-pip

RUN pip install Spandas && \
    pip install SigProfilerExtractor

RUN python3 -c 'from SigProfilerMatrixGenerator import install as genInstall; genInstall.install('GRCh38')'
